local function latexGrille(seed,n,bool)
  math.randomseed(seed)
  local fonctions = require("./TnT.lua")
  local grille = fonctions.genereGrille(n)
  local stringGrille =''
  local etat
  local cellule
  local caractere

  for i=1,n+1 do
    for j=1,n+1 do
      if i == 1 or j == 1 then
        etat = 'vide'
      else
        etat = 'pleine'
      end

      cellule = grille[i][j]

      if cellule == 'V' then
        caractere = ''
      elseif cellule == 'T' then
        if bool == true then
          caractere = '\\includegraphics[width=0.9\\dimensionscellule]{Tent.png}'
        else
          caractere = ''
        end
      elseif cellule == 'A' then
        caractere = '\\includegraphics[width=0.9\\dimensionscellule]{Tree.png}'
      else
        caractere = cellule
      end

      stringGrille = stringGrille .. '\\cellule{' .. etat .. '}{' .. caractere .. '}'
    end
      stringGrille = stringGrille .. '\\par\\vspace{-2\\epaisseurtrait}'
  end
  tex.sprint(stringGrille)
end




local seed = 1
local n = 18

latexGrille(seed,n,false)

tex.sprint('\\newpage')

latexGrille(seed,n,true)


\end{luacode*}
